         - git clone <repo name>

         - update README.md
         - git add README.md
         - git commit -m "updated README.md"
         - git push -u origin 6thJune_2023_Task1
         - Create a PR
         - If PR is approved, merge to main branch
         - git checkout main
         - git pull
